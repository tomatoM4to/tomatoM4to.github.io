CREATE OR REPLACE PROCEDURE BookInsertOrUpdate (
  p_book_id NUMBER,
  title VARCHAR2,
  author VARCHAR2,
  price NUMBER
)
AS
  mycount NUMBER; -- INTO: mycount라는 local변수에다가 바인딩함
BEGIN
  SELECT COUNT(*) INTO mycount FROM Book WHERE p_book_id = book_id;
  IF mycount = 0 THEN
    INSERT INTO Book VALUES (p_book_id, title, author, price);
  ELSE
    UPDATE Book SET title = title, author = author, price = price WHERE book_id = p_book_id;
  END IF;
END;

CREATE OR REPLACE PROCEDURE BookInsertOrUpdate (
  p_book_id IN NUMBER,         -- 입력 매개변수 IN 명시
  p_title IN VARCHAR2,
  p_author IN VARCHAR2,
  p_price IN NUMBER,
  p_result OUT VARCHAR2,       -- 처리 결과를 반환할 OUT 매개변수
  p_error_msg OUT VARCHAR2     -- 오류 메시지를 반환할 OUT 매개변수
)
AS
  v_count NUMBER;
BEGIN
  -- 초기화
  p_result := 'SUCCESS';
  p_error_msg := NULL;

  -- 도서 존재 여부 확인
  SELECT COUNT(*) INTO v_count
  FROM Book
  WHERE book_id = p_book_id;

  -- 도서가 없으면 INSERT
  IF v_count = 0 THEN
    INSERT INTO Book VALUES (p_book_id, p_title, p_author, p_price);
    p_result := 'INSERT_SUCCESS';

  -- 도서가 있으면 UPDATE
  ELSE
    UPDATE Book
    SET title = p_title,
        author = p_author,
        price = p_price
    WHERE book_id = p_book_id;
    p_result := 'UPDATE_SUCCESS';
  END IF;
END;

사용자 정의 함수

return 쓰는거 빼면 거의 비슷함, 이건 컴파일? 예기는 없고 그냥 실행하면 function 공간에 저장됀다 함

CREATE OR REPLACE FUNCTION fn (
  val number
) RETURN INT
IS
  val2 nNUMBER;
BEGIN
  IF val > 10000
    THEN val2 := val * 0.1;
  ELSE
    val2 := val * 0.05;
  END IF;
  RETURN val2;
END;

쓰는법, 특징은 프로시저처럼 EXEC가 필요 없이 바로 쓸수 있음, 사실상 내장함수(sum, avg) 같은것의 확장이기 때문에, 굉장히 유용함

SELECT fn(price) FROM Orders;

-- Book 테이블의 INSERT 트리거 생성
CREATE OR REPLACE TRIGGER afterInsertBook  -- 트리거 이름 정의
AFTER INSERT                              -- 트리거 발생 시점 (INSERT 이후)
ON Book                                   -- 트리거를 부착할 테이블
FOR EACH ROW                              -- 각 행마다 트리거 실행 (행 레벨 트리거)
DECLARE
    val NUMBER;                           -- 트리거 내에서 사용할 지역 변수
BEGIN
    -- Book 테이블에 INSERT가 발생할 때마다 로그 테이블에 기록
    -- :new는 방금 삽입된 새로운 행의 데이터를 참조
    INSERT INTO book_log VALUE(:new.book_id, ...);

    -- 트리거 실행 확인을 위한 메시지 출력
    DBMS_OUTPUT.PUT_LINE('INSERTED');
END;
